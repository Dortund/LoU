Clansmen_slot_1 = {
	generic = no             
	ai = yes                  
	has_country_shield = yes      
	slot = 1
	potential = {
		OR = {
			TAG = Z11
			FROM = {
				primary_culture = culture_clansmen
			}	
		}
	}
	the_isles_of_rast_arnor = {
		position = 1
		icon = mission_dunland_1 #Temp
		provinces_to_highlight = {
			area = rast_arnor_area
			NOT = {
				owned_by = ROOT
			}
		}
		trigger = {
			rast_arnor_area = {
				type = all
				owned_by = ROOT
			}
		}
		effect = {
			AND = {
			province_ID = {1406}
			add_province_modifier = {
				name = rast_arnor_lighthouse
				duration = -1
				desc = rast_arnor_lighthouse_tooltip
				}
			}
		}
	}
	rebuilding_the_harbor = {
		position = 2
		icon = mission_dunland_1 #Temp
		required_missions = {
			the_isles_of_rast_arnor
		}
		provinces_to_highlight = {
			AND = {
				province_ID = {153}
				NOT = {
					owned_by = ROOT
				}
			}
		}
		trigger = {
			AND = {
				153 = province_has_center_of_trade_of_level = 2
				153 = owned_by ROOT
			}
		}
		effect = { 
			AND = {
				Province_ID = {153}
				add_province_modifier = {
					name = rebuild_harbor
					duration = -1
					desc = rebuild_harbor_tootip
				}
			}
		}
	}
}
