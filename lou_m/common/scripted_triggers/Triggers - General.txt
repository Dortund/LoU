check_year = {
	is_year = $YEAR$
	NOT = {
		started_in = $DATE$
	}
}

	is_month = $MONTH$
	NOT = {
		started_in = $DATE$
	}
}

check_date = {
	is_year = $YEAR$
	is_month = $MONTH$
	NOT = {
		started_in = $STARTED_BEFORE$
	}
}
