set_against_race_opinions_effect = {
	if = {
		limit = {
			is_free_peoples_opinion = yes
		}
		every_country = {
			limit = {
				is_servants_of_evil_opinion = yes
				PREV = {
					NOT = {
						has_opinion_modifier = {
							who = PREV
							modifier = against_servants_of_darkness
						}
					}
				}
			}
		   
			# Good
			PREV = {
				add_opinion = {
					who = PREV # Evil
					modifier = against_servants_of_darkness
				}
			}
		   
			add_opinion = {
				who = PREV
				modifier = against_free_peoples
			}
		}
	}
}
