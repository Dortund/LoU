namespace = LOU_wights

# "Wights Roam The Land"
province_event = {
	id = LOU_wights.1
	title = "LOU_wights.1.t"
	desc = "LOU_wights.1.d"
	picture = COMET_SIGHTED_eventPicture

	trigger = {
        NOT = { has_province_modifier = province_wight_debuff }
        # Province
        OR = {
            culture = culture_wight
            religion = religion_wight
        }
        # Owner
        NOT = {
            owner = {
                primary_culture = culture_wight
                religion = religion_wight
            }
        }
	}
	
    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = "LOU_wights.1.a"

		add_province_modifier = {
			name = "province_wight_debuff"
			duration = -1
		}
	}
}
