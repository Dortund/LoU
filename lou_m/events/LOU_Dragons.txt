namespace = LOU_dragons

# "Dragons Roam The Skies"
province_event = {
	id = LOU_dragons.1
	title = "LOU_dragons.1.t"
	desc = "LOU_dragons.1.d"
	picture = COMET_SIGHTED_eventPicture

	trigger = {
        NOT = { has_province_modifier = province_dragon_debuff }
        # Province
        OR = {
            culture = culture_dragons
            religion = religion_dragons
        }
        # Owner
        NOT = {
            owner = {
                primary_culture = culture_dragons
                religion = religion_dragons
            }
        }
	}
	
    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = "LOU_dragons.1.a"

		add_province_modifier = {
			name = "province_dragon_debuff"
			duration = -1
		}
	}
}
