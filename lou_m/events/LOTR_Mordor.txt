			every_province = {
				limit = {
					owned_by = V18
					NOT = {
						is_in_capital_area = yes
					}

				add_core = V17

				limit = {

# Nurnoth joins Mordor
country_event = {
	id = flavor_mordor.3
	title = "flavor_mordor.3.t"
	desc = "flavor_mordor.3.d"
	picture = DIPLOMACY_eventPicture

	major = yes
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		tag = X86
	}

	immediate = {
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - flavor_mordor.3 - Option : immediate" }

		export_to_variable = {
			which = NumberOfStates
			value = trigger_value:num_of_states
			who = ROOT
		}

		divide_variable = {
			which = NumberOfStates
			value = 2
		}

		if = {
			limit = {
				any_owned_province = {
					OR = {
						area = maegond_vale_area
						area = thaurband_area
						area = sturlurtsa_nurn_area
					}
				}
			}
			every_owned_province = { 
				limit = {
					OR= {
						area = maegond_vale_area
						area = thaurband_area
						area = sturlurtsa_nurn_area
					}
				}
				add_core = V23
				cede_province = V23
			}
			set_variable = {
				which = effectId
				value = 0
			}
			V23 = {
				set_capital = 433
			}
		} 
		else_if = {
			limit = {
				num_of_states > 2
			}
			while = {
				limit = {
					check_variable = {
						which = NumberOfStates
						value > 2
					}
				}

				subtract_variable = {
					which = NumberOfStates
					value = 1
				}

				random_owned_area = { 
					add_core = V23
					cede_province = V23
				}
			}
			set_variable = {
				which = effectId
				value = 1
			}
			V23 = {
				set_capital = 433
			}
		}
		else = {
			switch_tag = V23
			set_variable = {
				which = effectId
				value = 2
			}
		}
		V17 = {
			vassalize = V23
		}
		V23 = {
			add_opinion = {
				who = V17
				modifier = nurnoth_mordor_relation
				years = 100
			}
			add_country_modifier = {
				name = mordor_nurnoth_subject
				duration = -1
			}
		}
	}
	
	option = {		# Oh no
		name = "flavor_mordor.3.a"
		
		trigger = {
			check_variable = {
				which = effectId
				value = 0
			}
		}
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - flavor_mordor.3 - Option : flavor_mordor.3.a" }

		custom_tooltip = nurnoth_spawns
	}

	option = {		# Oh no
		name = "flavor_mordor.3.b"
		
		trigger = {
			check_variable = {
				which = effectId
				value = 1
			}
		}
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - flavor_mordor.3 - Option : flavor_mordor.3.b" }

		custom_tooltip = nurnoth_spawns
	}

	option = {		# Oh no
		name = "flavor_mordor.3.c"
		trigger = {
			check_variable = {
				which = effectId
				value = 2
			}
		}
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - flavor_mordor.3 - Option : flavor_mordor.3.c" }

		tooltip = {
			switch_tag = V23
			V17 = {
				vassalize = V23
			}
		}
	}
}
