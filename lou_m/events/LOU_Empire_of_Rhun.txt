# Powerful Warlord
country_event = {
	id = LOU_empire_of_rhun.10
	title = LOU_empire_of_rhun.10.title
	desc = LOU_empire_of_rhun.10.desc
	picture = BAD_WITH_MONARCH_eventPicture
	
	trigger = {
        NOT = { total_development = 1000 }
        NOT = { is_subject = yes }
        NOT = { is_emperor_of_china = yes }
        OR = {
            capital_scope = {
                OR = {
                    region = rhun_region
                    region = kykurian_region
                    region = rhovanion_region
                    region = logathavuld_region
                }
            }
            event_target:EmperorOfChina = {
                is_neighbor_of = ROOT
            }
        }
        event_target:EmperorOfChina = {
            is_emperor_of_china = yes
        }
        OR = {
            religion = religion_ulgath
            religion = religion_easterling_men_of_darkness
        }
	}
    
    mean_time_to_happen = {
		months = 1200
        
        modifier = {
            factor = 2
            total_development = 500
        }
        modifier = {
            factor = 2
            total_development = 250
        }
        modifier = {
            factor = 0.5
            event_target:EmperorOfChina = {
                is_emperor_of_china = yes
                army_strength = {
                    who = ROOT
                    value = 1.5
                }
            }
        }
        modifier = {
            factor = 0.5
            event_target:EmperorOfChina = {
                is_emperor_of_china = yes
                army_strength = {
                    who = ROOT
                    value = 2
                }
            }
        }
        modifier = {
            factor = 0.5
            event_target:EmperorOfChina = {
                is_emperor_of_china = yes
                army_strength = {
                    who = ROOT
                    value = 3
                }
            }
        }
    }
    
    immediate = {
        hidden_effect = { save_event_target_as = tribute_country }
    }
    
	option = {
		name = LOU_empire_of_rhun.10.option.a

        event_target:EmperorOfChina = {
            create_subject = {
                subject_type = tributary_state
                subject = ROOT
            }
            country_event = {
                id = LOU_empire_of_rhun.11
            }
        }
        
        ai_chance = {
            factor = 75
        }
	}
    option = {
		name = LOU_empire_of_rhun.10.option.b

        add_prestige = 10
        
        ai_chance = {
            factor = 25
        }
	}
}
