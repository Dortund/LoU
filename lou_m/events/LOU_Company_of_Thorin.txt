#Current position of thorin
province_event = {
	id = LOU_company_of_thorin.1
	title = LOU_company_of_thorin.1.t
	desc = LOU_company_of_thorin.1.d
	picture = THORINS_COMPANY_eventPicture

	hidden = yes

	trigger = {
		NOT = {
			has_global_flag = company_of_thorin_lost
		}
		company_of_thorin = 100
		NOT = {
			province_id = 83
			has_province_modifier = thorin_passed
			has_province_modifier = thorin_position
		}
	}
	
	immediate = {
		if = { limit = { has_global_flag = enable_onaction_logging } log = "Event - province : [Root.GetName] LOU_company_of_thorin.1" }
		set_province_flag = thorin_set_position
		add_province_modifier = {
			name = "thorin_position"
			duration = -1
			hidden = yes
		}
		save_event_target_as = thorin_new_position
		every_neighbor_province = {
			limit = {
				company_of_thorin = 100
				has_province_modifier = thorin_position
				
				NOT = {
					province_id = ROOT
				}
			}
			
			add_province_modifier = {
				name = "thorin_passed"
				duration = -1
				hidden = yes
			}
			remove_province_modifier = thorin_position
			clr_province_flag = thorin_set_position
		}
	}

	option = {
		name = LOU_company_of_thorin.1.a
		
	}
}
