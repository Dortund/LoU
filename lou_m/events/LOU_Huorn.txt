namespace = LOU_huorn

# "Huorn Roam The Land"
province_event = {
	id = LOU_huorn.1
	title = "LOU_huorn.1.t"
	desc = "LOU_huorn.1.d"
	picture = BANKRUPTCY_eventPicture

	trigger = {
        NOT = { has_province_modifier = province_huorn_debuff }
        # Province
        OR = {
            culture = culture_huorn
            religion = religion_ents
        }
        # Owner
        NOT = {
            owner = {
                OR = {
                    primary_culture = culture_ents
                    primary_culture = culture_huorn
                }
                religion = religion_ents
            }
        }
	}
	
    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = "LOU_huorn.1.a"

		add_province_modifier = {
			name = "province_huorn_debuff"
			duration = -1
		}
	}
}
